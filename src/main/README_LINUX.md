# Commands
~~~bash
# start server
./bin/teaweb start

# stop server
./bin/teaweb stop

# restart server
./bin/teaweb restart

# print helps
./bin/teaweb -h

# print version
./bin/teaweb -v

# reset service state
./bin/teaweb reset
~~~
